document.getElementById('registroForm').addEventListener('submit', async function (e) { e.preventDefault(); const documento = document.getElementById('documento').value; const nombre = document.getElementById('nombre').value; const correo = document.getElementById('correo').value; const telefono = document.getElementById('telefono').value; const contrasena = document.getElementById('contrasena').value; const respuesta = document.getElementById('respuesta').value; const pregunta = document.getElementById('pregunta').value; const mensaje = document.getElementById('mensaje'); if (!pregunta) { mensaje.textContent = 'Por favor selecciona una pregunta de seguridad.'; return; } try { const res = await fetch('http://localhost:4545/usuario', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ t1: documento, t2: nombre, t3: correo, t4: telefono, t5: contrasena, pregunta, respuesta }), }); const data = await res.json(); if (res.ok || res.status === 201) { mensaje.textContent = '✅ Usuario registrado exitosamente'; mensaje.style.color = 'green'; if (data.token) { localStorage.setItem('token', data.token); } // Espera un segundo y redirige a perfil.html setTimeout(() => { window.location.href = 'perfil.html'; }, 1500); } else { mensaje.textContent = data?.error || data?.mensaje || 'Error al registrar usuario'; mensaje.style.color = 'red'; } } catch (err) { mensaje.textContent = 'Error de red. Intenta más tarde.'; } });
